FROM mcr.microsoft.com/devcontainers/javascript-node:1-22-bookworm

# Install OpenJDK 21 (required for current Firebase Emulators) from Debian Backports
RUN echo "deb http://deb.debian.org/debian bookworm-backports main" > /etc/apt/sources.list.d/backports.list && \
    apt-get update && \
    apt-get install -y --no-install-recommends -t bookworm-backports openjdk-21-jre-headless && \
    apt-get clean && \
    rm -rf /var/lib/apt/lists/*

# Install Firebase CLI globally
RUN npm install -g firebase-tools

# Expose ports for Node/Vite and Firebase Local Emulator Suite
EXPOSE 3000 4000 5000 5001 8080 8085 9000 9099 9199

# Default workspace directory
WORKDIR /workspace
